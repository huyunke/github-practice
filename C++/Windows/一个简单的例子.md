#include <windows.h>  
#include <tchar.h>  
  
  
static TCHAR szWindowClass[] = _T("DesktopApp");  
  
static TCHAR szTitle[] = _T("Windows Desktop");  //窗口名称
  
HINSTANCE hInst;  
  
LRESULT CALLBACK WndProc(HWND, UINT, WPARAM, LPARAM);  //声明WndProc函数
  
int WINAPI WinMain(  
   _In_ HINSTANCE hInstance,  
   _In_opt_ HINSTANCE hPrevInstance,  
   _In_ LPSTR lpCmdLine,  
   _In_ int nCmdShow  
) {  
   WNDCLASSEX wcex;  
  
   wcex.cbSize = sizeof(WNDCLASSEX);  
   wcex.style = CS_HREDRAW | CS_VREDRAW;  
   wcex.lpfnWndProc = WndProc;  
   wcex.cbClsExtra = 0;  
   wcex.cbWndExtra = 0;  
   wcex.hInstance = hInstance;  
   wcex.hIcon = LoadIcon(wcex.hInstance, IDI_APPLICATION);  
   wcex.hCursor = LoadCursor(NULL, IDC_ARROW);  
   wcex.hbrBackground = (HBRUSH) (COLOR_WINDOW + 1);  
   wcex.lpszMenuName = NULL;  
   wcex.lpszClassName = szWindowClass;  
   wcex.hIconSm = LoadIcon(wcex.hInstance, IDI_APPLICATION);  
  
   //注册窗口
   if (!RegisterClassEx(&wcex)) {  
      MessageBox(NULL,  
                 _T("Call to RegisterClassEx failed!"),  
                 _T("Windows Desktop"),  
                 NULL);  
  
      return 1;  
   }  
  
   hInst = hInstance;  
  
   //创建窗口
   HWND hWnd = CreateWindowEx(  
      WS_EX_OVERLAPPEDWINDOW,  
      szWindowClass,  
      szTitle,  
      WS_OVERLAPPEDWINDOW,  
      CW_USEDEFAULT, CW_USEDEFAULT,  
      500, 350,  
      NULL,  
      NULL,  
      hInstance,  
      NULL  
   );  
  
   if (!hWnd) {  
      MessageBox(NULL,  
                 _T("Call to CreateWindowEx failed!"),  
                 _T("Windows Desktop"),  
                 NULL);  
  
      return 1;  
   }  
  
   ShowWindow(hWnd, nCmdShow);  
   UpdateWindow(hWnd);  
  
   MSG msg;  
   while (GetMessage(&msg, NULL, 0, 0)) {  
      TranslateMessage(&msg);  
      DispatchMessage(&msg);  
   }  
  
   return (int) msg.wParam;  
}  
、
//回调函数
LRESULT CALLBACK WndProc(HWND hWnd, UINT message, WPARAM wParam, LPARAM lParam) {  
   PAINTSTRUCT ps;  
   HDC hdc;  //Handle to Device Context
   TCHAR greeting[] = _T("Hello, Windows desktop!");  
  
   switch (message) {  
      case WM_PAINT:  
         hdc = BeginPaint(hWnd, &ps);  //获取HDC
         
         RECT rect;  //定义一个“盒子”
         GetClientRect(hWnd, &rect);  //测量白板大小
  
         DrawText(hdc, greeting, -1, &rect, DT_CENTER | DT_VCENTER | DT_SINGLELINE);  
         EndPaint(hWnd, &ps);  //释放HDC，不能忘记
         break;  
      case WM_DESTROY:  
         PostQuitMessage(0);  
         break;  
      default:  
         return DefWindowProc(hWnd, message, wParam, lParam);  
   }  
  
   return 0;  
}

- 想象你是一个画家，去一个画室（Windows 系统）干活：

1. **画布 (Window)**：这是你要画画的地方。
    
2. **HDC (Device Context)**：这就是画室为你准备的**多功能画架**。
    
    - 画架上插着一支**画笔 (HPEN)**。
        
    - 画架上放着一个**调色盘 (HBRUSH)**。
        
    - 画架上挂着一张**画布 (HWND)**。
        
3. **绘图命令 (TextOut/DrawText)**：你告诉画架，“用现在的画笔在左上角写个字”。
